Function Main
    Declare Integer Max
    Declare Integer Min
    Declare Real Average
    Declare Integer Subject
    Declare Integer flag
    
    Assign Subject = getSubject()
    Declare Integer Array Marks[Subject]
    
    Assign flag = 0
    While flag < Subject
        Assign Marks[flag] = getMarks()
        Output Marks[flag]
        Assign flag = flag + 1
    End
    Assign Max = getMax(Marks, Subject)
    Assign Min = getMin(Marks , Subject)
    Assign Average = getAverage(Marks, Subject)
    Call getOutPut(Max, Min, Average)
End

Function getAverage (Integer Array Marks, Integer subject)
    Declare Integer flag
    Declare Real Average
    Declare Integer total
    
    Assign total = 0
    For flag = 0 to subject-1
        Assign total = total + Marks[flag]
    End
    Assign Average = total / subject
Return Integer Average

Function getMarks
    Declare Integer Marks
    
    Output "Please enter Grades :"
    Input Marks
Return Integer Marks

Function getMax (Integer Array Marks, Integer subject)
    Declare Integer flag
    Declare Integer Max
    
    Assign Max = 0
    For flag = 0 to subject-1
        If Max < Marks[flag]
            Assign Max = Marks[flag]
        End
    End
Return Integer Max

Function getMin (Integer Array Marks, Integer subject)
    Declare Integer flag
    Declare Integer Min
    
    Assign Min = Marks[1]
    For flag = 0 to subject-1
        If Min > Marks[flag]
            Assign Min = Marks[flag]
        End
    End
Return Integer Min

Function getOutPut (Integer Max, Integer Min, Integer Average)
    Output "Maximum Grade : "& Max
    Output "Minimum Grade : " & Min
    Output "Average Grade : " &Average
End

Function getSubject
    Declare Integer subject
    
    Output "Please Enter how many subjects you want to enter ?"
    Input subject
Return Integer subject
